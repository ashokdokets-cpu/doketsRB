    async function checkAuth() {
        const { data: { user } } = await supabase.auth.getUser();
        if (user) {
            currentUser = user;
            await loadUserProfile();
            return true;
        }
        return false;
    }

    async function loadUserProfile() {
        const { data } = await supabase
            .from('profiles')
            .select('*')
            .eq('id', currentUser.id)
            .single();
        userProfile = data;
        return data;
    }

    async function signUp(email, password, fullName) {
        const { data, error } = await supabase.auth.signUp({ email, password });
        if (error) return { error };
        await supabase.from('profiles').insert({ id: data.user.id, full_name: fullName, email: email });
        return { success: true };
    }

    async function logIn(email, password) {
        const { data, error } = await supabase.auth.signInWithPassword({ email, password });
        if (error) return { error };
        currentUser = data.user;
        await loadUserProfile();
        return { success: true };
    }

    async function logOut() {
        await supabase.auth.signOut();
        currentUser = null;
        userProfile = null;
        navigate('dashboard');
    }

    async function activatePlan(plan, paymentId) {
        const { data, error } = await supabase.rpc('update_user_plan', {
            p_user_id: currentUser.id,
            p_plan: plan,
            p_payment_id: paymentId,
        });
        if (error) return { error };
        await loadUserProfile();
        return data;
    }

    function parseAndMatch() {
        const jd = App.jobTarget.description.toLowerCase().trim();
        if (jd.length < 30) { alert('Please paste a complete job description (at least 30 characters).'); return; }
        const jt = { ...App.jobTarget, parsed: true };
        const skillMap = ['python','java','javascript','typescript','react','angular','vue','node','sql','mongodb','postgresql','aws','azure','docker','kubernetes','terraform','git','rest api','graphql','machine learning','data analysis','excel','tableau','power bi','figma','adobe','salesforce','hubspot','google analytics','seo','agile','scrum','jira','confluence','c++','c#','.net','php','django','flask','swift','kotlin','flutter','leadership','communication','project management','stakeholder management','strategic','analytical','problem-solving','team management','negotiation','mentoring'];
        jt.requiredSkills = []; jt.keywords = [];
        skillMap.forEach(skill => {
            const regex = new RegExp('\\b' + skill.replace(/[.*+?^${}()|[\]\\]/g,'\\$&') + '\\b', 'gi');
            const matches = jd.match(regex);
            if (matches) {
                if (/required|must have|essential/i.test(jd) && jd.indexOf(skill) > jd.search(/required|must have|essential/i)) jt.requiredSkills.push(skill);
                else jt.keywords.push(skill);
            }
        });
        jt.requiredSkills = [...new Set(jt.requiredSkills)]; jt.keywords = [...new Set(jt.keywords)];
        const tools = ['salesforce','hubspot','jira','confluence','slack','github','gitlab','figma','tableau','docker','kubernetes','aws','azure'];
        jt.tools = [...new Set(tools.filter(t => jd.includes(t)))];
        const expMatch = jd.match(/(\d+)\+?\s*(?:years?|yrs?)\s*(?:of\s*)?experience/i);
        jt.requiredExperience = expMatch ? parseInt(expMatch[1]) : 0;
        if (/ph\.?d|doctorate/i.test(jd)) jt.educationLevel = 'PhD';
        else if (/master|m\.s\.|m\.a\.|mba/i.test(jd)) jt.educationLevel = 'Masters';
        else if (/bachelor|b\.s\.|b\.a\.|undergrad/i.test(jd)) jt.educationLevel = 'Bachelors';
        const bullets = jd.match(/[•\-\*\d+\.]\s*(.{25,}?)(?=[•\-\*\d+\.\n]|$)/g);
        jt.keyResponsibilities = bullets ? bullets.map(b => b.replace(/^[•\-\*\d+\.]\s*/,'').trim()).filter(b => b.length > 10).slice(0,8) : [];
        updateState({ jobTarget: jt });

        const rd = App.resumeData;
        const allItems = [...new Set([...jt.requiredSkills, ...jt.keywords, ...jt.tools])];
        const candidateText = JSON.stringify(rd).toLowerCase();
        const candidateSkills = rd.skills.map(s => s.toLowerCase().trim());
        const match = { skillMatch: { matched: [], missing: [], score: 0 }, keywordCoverage: { present: [], absent: [], score: 0 }, experienceMatch: { score: 0, gap: '' }, educationMatch: { score: 0 }, overallScore: 0, autoSummary: '', autoBullets: {}, suggestions: [] };
        allItems.forEach(item => {
            if (candidateSkills.some(s => s.includes(item) || item.includes(s)) || candidateText.includes(item)) match.skillMatch.matched.push(item);
            else match.skillMatch.missing.push(item);
        });
        match.skillMatch.score = allItems.length ? Math.round((match.skillMatch.matched.length/allItems.length)*100) : 100;
        const allKW = [...new Set([...jt.keywords, ...jt.tools])];
        allKW.forEach(kw => { if (candidateText.includes(kw.toLowerCase())) match.keywordCoverage.present.push(kw); else match.keywordCoverage.absent.push(kw); });
        match.keywordCoverage.score = allKW.length ? Math.round((match.keywordCoverage.present.length/allKW.length)*100) : 100;
        let totalYears = 0; rd.experience.forEach(exp => { const yrs = (exp.dates||'').match(/(\d{4})/g); if (yrs && yrs.length>=2) totalYears += Math.max(0,parseInt(yrs[1])-parseInt(yrs[0])); });
        match.experienceMatch.score = jt.requiredExperience ? (totalYears>=jt.requiredExperience?90:totalYears>=jt.requiredExperience*0.7?65:30) : (totalYears>0?80:20);
        match.experienceMatch.gap = totalYears>0?`${totalYears}+ years`:'No experience listed';
        const levels={phd:4,masters:3,bachelors:2}; const reqLvl=levels[jt.educationLevel?.toLowerCase()]||0;
        const eduText=JSON.stringify(rd.education).toLowerCase(); let maxLvl=0;
        Object.entries(levels).forEach(([k,v])=>{if(eduText.includes(k))maxLvl=Math.max(maxLvl,v);});
        match.educationMatch.score=reqLvl?(maxLvl>=reqLvl?95:maxLvl>=reqLvl-1?60:25):(rd.education.length>0?85:10);
        if(rd.personal.fullName&&jt.title){match.autoSummary=`Results-driven ${jt.title} with ${totalYears}+ years experience in ${match.skillMatch.matched.slice(0,3).join(', ')||'delivering outcomes'}.`;}
        rd.experience.forEach((exp,i)=>{const skill=match.skillMatch.matched[i%Math.max(1,match.skillMatch.matched.length)];if(skill&&!(exp.bullets||'').toLowerCase().includes(skill.toLowerCase()))match.autoBullets[i]=`• Applied ${skill} to ${(jt.keyResponsibilities[i%Math.max(1,jt.keyResponsibilities.length)]||'key outcomes').toLowerCase()}, achieving measurable impact`;});
        if(match.skillMatch.missing.length>0)match.suggestions.push(`⚠️ Missing ${match.skillMatch.missing.length} skills: ${match.skillMatch.missing.slice(0,6).join(', ')}`);
        if(match.keywordCoverage.absent.length>0)match.suggestions.push(`🔑 Add keywords: ${match.keywordCoverage.absent.slice(0,5).join(', ')}`);
        match.overallScore=Math.round(match.skillMatch.score*0.35+match.keywordCoverage.score*0.30+match.experienceMatch.score*0.20+match.educationMatch.score*0.15);
        updateState({ jobMatch: match });
    }

    function getATSScore(){const rd=App.resumeData;const s={contact:rd.personal.fullName&&rd.personal.email?95:rd.personal.fullName?60:20,summary:rd.summary?rd.summary.length>80?90:65:10,exp:rd.experience.length>=2?90:rd.experience.length===1?60:15,edu:rd.education.length>0?85:20,skills:rd.skills.length>=8?90:rd.skills.length>=5?70:rd.skills.length>0?45:10};return{total:Math.round(Object.values(s).reduce((a,b)=>a+b,0)/5),breakdown:s};}
    function getCompositeScore(){const ats=getATSScore().total;const jm=App.jobMatch.overallScore||0;return jm>0?Math.round(ats*0.4+jm*0.6):ats;}
